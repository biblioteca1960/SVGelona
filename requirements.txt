✅ Installation Verification
Create a file check_install.py:

python
#!/usr/bin/env python3
"""
Check if all dependencies are installed correctly
"""

import sys

def check_import(module_name, package_name=None):
    """Check if a module can be imported"""
    if package_name is None:
        package_name = module_name
    try:
        __import__(module_name)
        print(f"✅ {package_name} installed")
        return True
    except ImportError as e:
        print(f"❌ {package_name} NOT installed: {e}")
        return False

def main():
    """Check all dependencies"""
    print("="*60)
    print("SVGelona - Dependency Check")
    print("="*60)
    
    # Core dependencies
    print("\n📦 CORE DEPENDENCIES:")
    check_import("flask")
    check_import("numpy")
    
    # RAG System
    print("\n🔍 RAG SYSTEM:")
    check_import("sentence_transformers", "sentence-transformers")
    check_import("sklearn", "scikit-learn")
    
    # Language detection
    print("\n🌐 LANGUAGE DETECTION:")
    check_import("langdetect")
    
    # HTTP
    print("\n🌍 HTTP REQUESTS:")
    check_import("requests")
    
    # PDF extraction
    print("\n📄 PDF EXTRACTION:")
    check_import("PyPDF2")
    check_import("pdfplumber")
    check_import("textract")
    
    # Optional development tools
    print("\n🛠️ DEVELOPMENT TOOLS (optional):")
    check_import("pytest")
    check_import("black")
    check_import("flake8")
    check_import("mypy")
    
    print("\n" + "="*60)
    print("Check complete!")
    print("="*60)

if __name__ == "__main__":
    main()

Docker Support
If you prefer Docker, here's a minimal Dockerfile:

dockerfile
FROM python:3.9-slim

WORKDIR /app

# Install system dependencies for pdf extraction
RUN apt-get update && apt-get install -y \
    poppler-utils \
    antiword \
    tesseract-ocr \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements and install
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application
COPY . .

# Run the application
CMD ["python", "narg.py"]
